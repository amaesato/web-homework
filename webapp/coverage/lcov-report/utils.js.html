export const getHeadersAndRows = (data = []) => {
  const [headers, rows] = data.reduce(([headers, rows], item) => {
    const keys = Object.keys(item).filter(k => k !== '__typename')
    if (!headers.length) {
      headers = [...keys]
    }
    rows.push(keys.map(key => item[key]))
    return [headers, rows]
  }, [[], []])
  return [headers, rows]
}
getHeadersAndRows.propTypes = arrayOf(TransactionT)
